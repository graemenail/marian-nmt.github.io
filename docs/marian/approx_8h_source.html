8 // approximate any unary float function within range with
9 // piecewise linear functions in equal steps.
10 //
11 // Example:
12 // static Approx<10, 0, 100> approxSigmoid(stableSigmoid);
13 // float y = approxSigmoid(x);
14 //
15 // Creates a functor for range [-10,10] with piecewise linear
16 // approximations of a sigmoid, 100 pieces, step 0.2.
17 // This is quite fast on the CPU.
18 //
19 // approxSigmoid.grad(x) computes the corresponding gradient.
20 //
21 // When used as a local variable, use static keyword to create
22 // only once.

31  Approx(const Function& function) {
32  for(int i = 1; i <= pieces; ++i) {
33  float x0 = domain(i - 1);
34  float x1 = domain(i);
35  float y0 = function(x0);
36  float y1 = function(x1);
37 
38  a[i] = (y1 - y0) / (x1 - x0);
39  b[i] = y0 - a[i] * x0;
