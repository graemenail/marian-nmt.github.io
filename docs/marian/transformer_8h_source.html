11  Expr AddPositionalEmbeddings(Ptr<ExpressionGraph> graph,
12  Expr input,
13  int start = 0) {
14  using namespace keywords;
15 
16  int dimEmb = input->shape()[-1];
17  int dimWords = input->shape()[-3];
18 
19  float num_timescales = dimEmb / 2;
20  float log_timescale_increment = std::log(10000.f) / (num_timescales - 1.f);
21 
22  std::vector<float> vPos(dimEmb * dimWords, 0);
23  for(int p = start; p < dimWords + start; ++p) {
24  for(int i = 0; i < num_timescales; ++i) {
25  float v = p * std::exp(i * -log_timescale_increment);
26  vPos[(p - start) * dimEmb + i] = std::sin(v);
27  vPos[(p - start) * dimEmb + num_timescales + i] = std::cos(v);
28  }
29  }
30 
31  // shared across batch entries
32  auto signal = graph->constant({dimWords, 1, dimEmb},
33  init = inits::from_vector(vPos));

85  std::string prefix,
86  std::string ops,
87  Expr input,
88  float dropProb = 0.0f) {
89  using namespace keywords;
90 
91  int dimModel = input->shape()[-1];
92  auto output = input;
93  for(auto op : ops) {
94  // dropout
95  if(op == 'd' && dropProb > 0.0f) {
96  auto dropMask = graph->dropout(dropProb, output->shape());
97  output = dropout(output, mask = dropMask);
98  }
99  // layer normalization
100  if(op == 'n') {
101  auto scale = graph->param(
102  prefix + "_ln_scale_pre", {1, dimModel}, init = inits::ones);
103  auto bias = graph->param(
104  prefix + "_ln_bias_pre", {1, dimModel}, init = inits::zeros);
105  output = layer_norm(output, scale, bias);
106  }

111  Expr PostProcess(Ptr<ExpressionGraph> graph,
112  std::string prefix,
113  std::string ops,
114  Expr input,
115  Expr prevInput,
116  float dropProb = 0.0f) {
117  using namespace keywords;
118 
119  int dimModel = input->shape()[-1];
120  auto output = input;
121  for(auto op : ops) {
122  // dropout
123  if(op == 'd' && dropProb > 0.0f) {
124  auto dropMask = graph->dropout(dropProb, output->shape());
125  output = dropout(output, mask = dropMask);
126  }
127  // skip connection, moved behind layer normalization
128  if(op == 'a') {
129  output = output + prevInput;
130  }
131  // highway connection
132  if(op == 'h') {
133  auto Wh = graph->param(
134  prefix + "_Wh", {dimModel, dimModel}, init = inits::glorot_uniform);
135  auto bh
136  = graph->param(prefix + "_bh", {1, dimModel}, init = inits::zeros);
137 
138  auto t = affine(prevInput, Wh, bh);
139  output = highway(output, prevInput, t);
140  }
141  // layer normalization
142  if(op == 'n') {
143  auto scale = graph->param(
144  prefix + "_ln_scale", {1, dimModel}, init = inits::ones);
145  auto bias = graph->param(
146  prefix + "_ln_bias", {1, dimModel}, init = inits::zeros);
147  output = layer_norm(output, scale, bias, 1e-6);
148  }
149  }
150  return output;
151  }

153  Expr Attention(Ptr<ExpressionGraph> graph,
154  Ptr<Options> options,
155  std::string prefix,
156  Expr q,
157  Expr k,
158  Expr v,
159  Expr mask = nullptr,
160  bool inference = false) {
161  using namespace keywords;
162 
163  float dk = k->shape()[-1];
164 
165  // scaling to avoid extreme values due to matrix multiplication
166  float scale = 1.0 / std::sqrt(dk);
167 
168  // softmax over batched dot product of query and keys (applied over all
169  // time steps and batch entries), also add mask for illegal connections
170 
171  // @TODO: do this better
172  int dimBeamQ = q->shape()[-4];
173  int dimBeamK = k->shape()[-4];
174  int dimBeam = dimBeamQ / dimBeamK;
175  if(dimBeam > 1) {
176  k = repeat(k, dimBeam, axis = -4);
177  v = repeat(v, dimBeam, axis = -4);
178  }
179 
180  auto weights = softmax(bdot(q, k, false, true, scale) + mask);
181 
182  // optional dropout for attention weights
183  float dropProb
184  = inference ? 0 : options->get<float>("transformer-dropout-attention");
185  if(dropProb) {
186  auto dropMask = graph->dropout(dropProb, weights->shape());
187  weights = dropout(weights, mask = dropMask);
188  }
189 
190  // apply attention weights to values
191  return bdot(weights, v);
192  }

194  Expr MultiHead(Ptr<ExpressionGraph> graph,
195  Ptr<Options> options,
196  std::string prefix,
197  int dimOut,
198  int dimHeads,
199  Expr q,
200  const std::vector<Expr> &keys,
201  const std::vector<Expr> &values,
202  const std::vector<Expr> &masks,
203  bool inference = false) {
204  using namespace keywords;
205 
206  int dimModel = q->shape()[-1];
207 
208  auto Wq = graph->param(
209  prefix + "_Wq", {dimModel, dimModel}, init = inits::glorot_uniform);
210  auto bq = graph->param(prefix + "_bq", {1, dimModel}, init = inits::zeros);
211  auto qh = affine(q, Wq, bq);
212  qh = SplitHeads(qh, dimHeads);
213 
214  std::vector<Expr> outputs;
215  for(int i = 0; i < keys.size(); ++i) {
216  std::string prefixProj = prefix;
217  if(i > 0)
218  prefixProj += "_enc" + std::to_string(i + 1);
219 
220  auto Wk = graph->param(prefixProj + "_Wk",
221  {dimModel, dimModel},
222  init = inits::glorot_uniform);
223  auto bk = graph->param(
224  prefixProj + "_bk", {1, dimModel}, init = inits::zeros);
225 
226  auto Wv = graph->param(prefixProj + "_Wv",
227  {dimModel, dimModel},
228  init = inits::glorot_uniform);
229  auto bv = graph->param(
230  prefixProj + "_bv", {1, dimModel}, init = inits::zeros);
231 
232  auto kh = affine(keys[i], Wk, bk);
233  auto vh = affine(values[i], Wv, bv);
234 
235  kh = SplitHeads(kh, dimHeads);
236  vh = SplitHeads(vh, dimHeads);
237 
238  // apply multi-head attention to downscaled inputs
239  auto output
240  = Attention(graph, options, prefix, qh, kh, vh, masks[i], inference);
241 
242  output = JoinHeads(output, q->shape()[-4]);
243 
244  outputs.push_back(output);
245  }
246 
247  Expr output;
248  if(outputs.size() > 1)
249  output = concatenate(outputs, axis = -1);
250  else
251  output = outputs.front();
252 
253  int dimAtt = output->shape()[-1];
254 
255  auto Wo = graph->param(
256  prefix + "_Wo", {dimAtt, dimOut}, init = inits::glorot_uniform);
257  auto bo = graph->param(prefix + "_bo", {1, dimOut}, init = inits::zeros);
258  output = affine(output, Wo, bo);
259 
260  return output;
261  }

384  Ptr<EncoderState> build(Ptr<ExpressionGraph> graph,
385  Ptr<data::CorpusBatch> batch) {
386  using namespace keywords;
387 
388  int dimEmb = opt<int>("dim-emb");
389  int dimBatch = batch->size();
390  int dimSrcWords = (*batch)[batchIndex_]->batchWidth();
391 
392  auto embeddings = WordEmbeddings(graph, batch);
393 
394  // select embeddings that occur in the batch
395  Expr batchEmbeddings, batchMask;
396  std::tie(batchEmbeddings, batchMask)
397  = EncoderBase::lookup(embeddings, batch);
398 
399  // apply dropout over source words
400  float dropoutSrc = inference_ ? 0 : opt<float>("dropout-src");
401  if(dropoutSrc) {
402  int srcWords = batchEmbeddings->shape()[-3];
403  auto dropMask = graph->dropout(dropoutSrc, {srcWords, 1, 1});
404  batchEmbeddings = dropout(batchEmbeddings, mask = dropMask);
405  }
406 
407  // according to paper embeddings are scaled by \sqrt(d_m)
408  auto scaledEmbeddings = std::sqrt(dimEmb) * batchEmbeddings;
409  scaledEmbeddings = AddPositionalEmbeddings(graph, scaledEmbeddings);
410 
411  // reorganize batch and timestep
412  scaledEmbeddings = atleast_nd(scaledEmbeddings, 4);
413  batchMask = atleast_nd(batchMask, 4);
414  auto layer = TransposeTimeBatch(scaledEmbeddings);
415  auto layerMask
416  = reshape(TransposeTimeBatch(batchMask), {1, dimBatch, 1, dimSrcWords});
417 
418  auto opsEmb = opt<std::string>("transformer-postprocess-emb");
419 
420  float dropProb = inference_ ? 0 : opt<float>("transformer-dropout");
421  layer = PreProcess(graph, prefix_ + "_emb", opsEmb, layer, dropProb);
422 
423  layerMask = InverseMask(layerMask);
424 
425  // apply layers
426  for(int i = 1; i <= opt<int>("enc-depth"); ++i) {
427  layer = LayerAttention(graph,
428  options_,
429  prefix_ + "_l" + std::to_string(i) + "_self",
430  layer,
431  layer,
432  layer,
433  layerMask,
434  inference_);
435 
436  layer = LayerFFN(graph,
437  options_,
438  prefix_ + "_l" + std::to_string(i) + "_ffn",
439  layer,
440  inference_);
441  }
442 
443  // restore organization of batch and time steps. This is currently required
444  // to make RNN-based decoders and beam search work with this. We are looking
445  // into makeing this more natural.

504  if(dropoutTrg) {
505  int trgWords = embeddings->shape()[-3];
506  auto trgWordDrop = graph->dropout(dropoutTrg, {trgWords, 1, 1});
507  embeddings = dropout(embeddings, mask = trgWordDrop);
508  }
509

512  int dimEmb = embeddings->shape()[-1];

514  if(embeddings->shape().size() > 3)
515  dimBeam = embeddings->shape()[-4];
516 
517  // according to paper embeddings are scaled by \sqrt(d_m)
518  auto scaledEmbeddings = std::sqrt(dimEmb) * embeddings;
519 
520  int startPos = 0;
521  auto prevDecoderStates = state->getStates();
522  if(prevDecoderStates.size() > 0)
523  startPos = prevDecoderStates[0].output->shape()[-2];
524

526  = AddPositionalEmbeddings(graph, scaledEmbeddings, startPos);

532 
533  auto opsEmb = opt<std::string>("transformer-postprocess-emb");
534  float dropProb = inference_ ? 0 : opt<float>("transformer-dropout");

536  query = PreProcess(graph, prefix_ + "_emb", opsEmb, query, dropProb);

538  rnn::States decoderStates;
539  int dimTrgWords = query->shape()[-2];
540  int dimBatch = query->shape()[-3];

542  if(decoderMask) {

544  decoderMask = reshape(TransposeTimeBatch(decoderMask),
545  {1, dimBatch, 1, dimTrgWords});

549  }

592  inference_);
