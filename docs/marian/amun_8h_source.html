158  float dropoutTrg = inference_ ? 0 : options_->get<float>("dropout-trg");

161  auto embeddings = stateAmun->getTargetEmbeddings();
162 
163  if(dropoutTrg) {
164  int trgWords = embeddings->shape()[2];
165  auto trgWordDrop = graph->dropout(dropoutTrg, {1, 1, trgWords});
166  embeddings = dropout(embeddings, mask=trgWordDrop);
167  }
168 
169  if(!attention_)
170  attention_ = New<GlobalAttention>("decoder",
171  state->getEncoderState(),
172  dimDecState,
173  dropout_prob=dropoutRnn,
174  normalize=layerNorm);
175 
176  if(!rnn)
177  rnn = New<RNN<CGRU>>(graph, "decoder",
178  dimTrgEmb, dimDecState,
179  attention_,
180  dropout_prob=dropoutRnn,
181  normalize=layerNorm);
182  auto stateOut = (*rnn)(embeddings, stateAmun->getState());
183 
184  bool single = stateAmun->doSingleStep();
185 
186  auto alignedContextsVec = attention_->getContexts();
187  auto alignedContext = single ?
188  alignedContextsVec.back() :
189  concatenate(alignedContextsVec, keywords::axis=2);
190 
192  auto logitsL1 = Dense("ff_logit_l1", dimTrgEmb,
193  activation=act::tanh,
194  normalize=layerNorm)
195  (embeddings, stateOut, alignedContext);
196 
197  auto logitsOut = Dense("ff_logit_l2", dimTrgVoc)(logitsL1);
198 
199  return New<DecoderStateAmun>(stateOut, logitsOut,
200  state->getEncoderState());
201  }
