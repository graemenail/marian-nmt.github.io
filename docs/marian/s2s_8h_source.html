15  // build two separate stacks, concatenate top output
16  first = opt<int>("enc-depth");
17  second = 0;
18  } else {
19  // build 1-layer bidirectional stack, concatenate,
20  // build n-1 layer unidirectional stack
21  first = 1;
22  second = opt<int>("enc-depth") - first;
23  }
24 
25  auto forward = type == "alternating" ? rnn::dir::alternating_forward
26  : rnn::dir::forward;
27 
28  auto backward = type == "alternating" ? rnn::dir::alternating_backward
29  : rnn::dir::backward;
30 
31  using namespace keywords;
32  float dropoutRnn = inference_ ? 0 : opt<float>("dropout-rnn");
33 
34  auto rnnFw = rnn::rnn(graph) //
35  ("type", opt<std::string>("enc-cell")) //
36  ("direction", forward) //
37  ("dimInput", embeddings->shape()[-1]) //
38  ("dimState", opt<int>("dim-rnn")) //
39  ("dropout", dropoutRnn) //
40  ("layer-normalization", opt<bool>("layer-normalization")) //
41  ("skip", opt<bool>("skip"));
42 
43  for(int i = 1; i <= first; ++i) {
44  auto stacked = rnn::stacked_cell(graph);
45  for(int j = 1; j <= opt<int>("enc-cell-depth"); ++j) {
46  std::string paramPrefix = prefix_ + "_bi";
47  if(i > 1)
48  paramPrefix += "_l" + std::to_string(i);
49  if(i > 1 || j > 1)
50  paramPrefix += "_cell" + std::to_string(j);
51  bool transition = (j > 1);
52 
53  stacked.push_back(rnn::cell(graph) //
54  ("prefix", paramPrefix) //
55  ("transition", transition));
56  }
57  rnnFw.push_back(stacked);
58  }
59

61  ("type", opt<std::string>("enc-cell")) //
62  ("direction", backward) //
63  ("dimInput", embeddings->shape()[-1]) //
64  ("dimState", opt<int>("dim-rnn")) //
65  ("dropout", dropoutRnn) //
66  ("layer-normalization", opt<bool>("layer-normalization")) //
67  ("skip", opt<bool>("skip"));
68 
69  for(int i = 1; i <= first; ++i) {
70  auto stacked = rnn::stacked_cell(graph);
71  for(int j = 1; j <= opt<int>("enc-cell-depth"); ++j) {
72  std::string paramPrefix = prefix_ + "_bi_r";
73  if(i > 1)
74  paramPrefix += "_l" + std::to_string(i);
75  if(i > 1 || j > 1)
76  paramPrefix += "_cell" + std::to_string(j);
77  bool transition = (j > 1);
78 
79  stacked.push_back(rnn::cell(graph) //
80  ("prefix", paramPrefix) //
81  ("transition", transition));
82  }
83  rnnBw.push_back(stacked);
84  }
85 
86  auto context = concatenate({rnnFw->transduce(embeddings, mask),
87  rnnBw->transduce(embeddings, mask)},
88  axis = -1);
89 
90  if(second > 0) {
91  // add more layers (unidirectional) by transducing the output of the
92  // previous bidirectional RNN through multiple layers
93 
94  // construct RNN first
95  auto rnnUni = rnn::rnn(graph) //
96  ("type", opt<std::string>("enc-cell")) //
97  ("dimInput", 2 * opt<int>("dim-rnn")) //
98  ("dimState", opt<int>("dim-rnn")) //
99  ("dropout", dropoutRnn) //
100  ("layer-normalization", opt<bool>("layer-normalization")) //
101  ("skip", opt<bool>("skip"));
102 
103  for(int i = first + 1; i <= second + first; ++i) {
104  auto stacked = rnn::stacked_cell(graph);
105  for(int j = 1; j <= opt<int>("enc-cell-depth"); ++j) {
106  std::string paramPrefix = prefix_ + "_l" + std::to_string(i) + "_cell"
107  + std::to_string(j);
108  stacked.push_back(rnn::cell(graph)("prefix", paramPrefix));
109  }
110  rnnUni.push_back(stacked);
111  }
112 
113  // transduce context to new context
114  context = rnnUni->transduce(context);
115  }
116  return context;

179  Ptr<rnn::RNN> constructDecoderRNN(Ptr<ExpressionGraph> graph,
180  Ptr<DecoderState> state) {
181  float dropoutRnn = inference_ ? 0 : opt<float>("dropout-rnn");
182  auto rnn = rnn::rnn(graph) //
183  ("type", opt<std::string>("dec-cell")) //
184  ("dimInput", opt<int>("dim-emb")) //
185  ("dimState", opt<int>("dim-rnn")) //
186  ("dropout", dropoutRnn) //
187  ("layer-normalization", opt<bool>("layer-normalization")) //
188  ("nematus-normalization",
189  options_->has("original-type")
190  && opt<std::string>("original-type") == "nematus") //
191  ("skip", opt<bool>("skip"));
192 
193  size_t decoderLayers = opt<size_t>("dec-depth");
194  size_t decoderBaseDepth = opt<size_t>("dec-cell-base-depth");
195  size_t decoderHighDepth = opt<size_t>("dec-cell-high-depth");
196 
197  // setting up conditional (transitional) cell
198  auto baseCell = rnn::stacked_cell(graph);
199  for(int i = 1; i <= decoderBaseDepth; ++i) {
200  bool transition = (i > 2);
201  auto paramPrefix = prefix_ + "_cell" + std::to_string(i);
202  baseCell.push_back(rnn::cell(graph) //
203  ("prefix", paramPrefix) //
204  ("final", i > 1) //
205  ("transition", transition));
206  if(i == 1) {
207  for(int k = 0; k < state->getEncoderStates().size(); ++k) {
208  auto attPrefix = prefix_;
209  if(state->getEncoderStates().size() > 1)
210  attPrefix += "_att" + std::to_string(k + 1);
211 
212  auto encState = state->getEncoderStates()[k];
213 
214  baseCell.push_back(
215  rnn::attention(graph)("prefix", attPrefix).set_state(encState));
216  }
217  }
218  }
219  // Add cell to RNN (first layer)
220  rnn.push_back(baseCell);
221 
222  // Add more cells to RNN (stacked RNN)
223  for(int i = 2; i <= decoderLayers; ++i) {
224  // deep transition
225  auto highCell = rnn::stacked_cell(graph);
226 
227  for(int j = 1; j <= decoderHighDepth; j++) {
228  auto paramPrefix
229  = prefix_ + "_l" + std::to_string(i) + "_cell" + std::to_string(j);
230  highCell.push_back(rnn::cell(graph)("prefix", paramPrefix));
231  }
232 
233  // Add cell to RNN (more layers)
234  rnn.push_back(highCell);
235  }
236 
237  return rnn.construct();
238  }

243  virtual Ptr<DecoderState> startState(
244  Ptr<ExpressionGraph> graph,
245  Ptr<data::CorpusBatch> batch,
246  std::vector<Ptr<EncoderState>>& encStates) {
247  using namespace keywords;
248 
249  std::vector<Expr> meanContexts;
250  for(auto& encState : encStates) {
251  // average the source context weighted by the batch mask
252  // this will remove padded zeros from the average
253  meanContexts.push_back(weighted_average(
254  encState->getContext(), encState->getMask(), axis = -3));
255  }
256 
257  Expr start;
258  if(!meanContexts.empty()) {
259  // apply single layer network to mean to map into decoder space
260  auto mlp = mlp::mlp(graph).push_back(
261  mlp::dense(graph) //
262  ("prefix", prefix_ + "_ff_state") //
263  ("dim", opt<int>("dim-rnn")) //
264  ("activation", mlp::act::tanh) //
265  ("layer-normalization", opt<bool>("layer-normalization")) //
266  ("nematus-normalization",
267  options_->has("original-type")
268  && opt<std::string>("original-type") == "nematus") //
269  );
270  start = mlp->apply(meanContexts);
271  } else {
272  int dimBatch = batch->size();
273  int dimRnn = opt<int>("dim-rnn");
274 
275  start = graph->constant({dimBatch, dimRnn}, init = inits::zeros);
276  }
277 
278  rnn::States startStates(opt<size_t>("dec-depth"), {start, start});
279  return New<DecoderState>(startStates, nullptr, encStates);

282  virtual Ptr<DecoderState> step(Ptr<ExpressionGraph> graph,
283  Ptr<DecoderState> state) {
284  using namespace keywords;
285 
286  auto embeddings = state->getTargetEmbeddings();
287 
288  // dropout target words
289  float dropoutTrg = inference_ ? 0 : opt<float>("dropout-trg");
290  if(dropoutTrg) {
291  int trgWords = embeddings->shape()[-3];
292  auto trgWordDrop = graph->dropout(dropoutTrg, {trgWords, 1, 1});
293  embeddings = dropout(embeddings, mask = trgWordDrop);
294  }
295 
296  if(!rnn_)
297  rnn_ = constructDecoderRNN(graph, state);
298 
299  // apply RNN to embeddings, initialized with encoder context mapped into
300  // decoder space
301  auto decoderContext = rnn_->transduce(embeddings, state->getStates());
302 
303  // retrieve the last state per layer. They are required during translation
304  // in order to continue decoding for the next word
305  rnn::States decoderStates = rnn_->lastCellStates();
306 
307  std::vector<Expr> alignedContexts;
308  for(int k = 0; k < state->getEncoderStates().size(); ++k) {
309  // retrieve all the aligned contexts computed by the attention mechanism
310  auto att = rnn_->at(0)
311  ->as<rnn::StackedCell>()
312  ->at(k + 1)
313  ->as<rnn::Attention>();
314  alignedContexts.push_back(att->getContext());
315  }
316 
317  Expr alignedContext;
318  if(alignedContexts.size() > 1)
319  alignedContext = concatenate(alignedContexts, axis = -1);
320  else if(alignedContexts.size() == 1)
321  alignedContext = alignedContexts[0];
322 
323  // construct deep output multi-layer network layer-wise
324  auto layer1 = mlp::dense(graph) //
325  ("prefix", prefix_ + "_ff_logit_l1") //
326  ("dim", opt<int>("dim-emb")) //
327  ("activation", mlp::act::tanh) //
328  ("layer-normalization", opt<bool>("layer-normalization")) //
329  ("nematus-normalization",
330  options_->has("original-type")
331  && opt<std::string>("original-type") == "nematus");
332 
333  int dimTrgVoc = opt<std::vector<int>>("dim-vocabs")[batchIndex_];
334 
335  auto layer2 = mlp::dense(graph) //
336  ("prefix", prefix_ + "_ff_logit_l2") //
337  ("dim", dimTrgVoc);
338 
339  if(opt<bool>("tied-embeddings") || opt<bool>("tied-embeddings-all")) {
340  std::string tiedPrefix = prefix_ + "_Wemb";
341  if(opt<bool>("tied-embeddings-all") || opt<bool>("tied-embeddings-src"))
342  tiedPrefix = "Wemb";
343  layer2.tie_transposed("W", tiedPrefix);
344  }
345 
346  // assemble layers into MLP and apply to embeddings, decoder context and
347  // aligned source context
348  auto output = mlp::mlp(graph) //
349  .push_back(layer1) //
350  .push_back(layer2);
351 
352  Expr logits;
353  if(alignedContext)
354  logits = output->apply(embeddings, decoderContext, alignedContext);
355  else
356  logits = output->apply(embeddings, decoderContext);
357 
358  // return unormalized(!) probabilities
359  return New<DecoderState>(decoderStates, logits, state->getEncoderStates());
360  }
