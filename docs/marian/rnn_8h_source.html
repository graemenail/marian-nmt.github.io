56  States apply(const Expr input,
57  const States initialState,
58  const Expr mask = nullptr) {
59  last_.clear();
60 
61  State state = initialState.front();
62 
63  cell_->clear();
64 
65  auto xWs = cell_->applyInput({input});
66 
67  size_t timeSteps = input->shape()[-3];
68 
69  States outputs;
70  for(size_t i = 0; i < timeSteps; ++i) {
71  int j = i;
72 
73  if(direction_ == dir::backward)
74  j = timeSteps - i - 1;
75 
76  std::vector<Expr> steps(xWs.size());
77  std::transform(xWs.begin(), xWs.end(), steps.begin(), [j](Expr e) {
78  return step(e, j, -3);
79  });
80 
81  if(mask)
82  state = cell_->applyState(steps, state, step(mask, j, -3));
83  else
84  state = cell_->applyState(steps, state);
85 
86  outputs.push_back(state);
87  }
88 
89  if(direction_ == dir::backward)
90  outputs.reverse();
91 
92  last_.push_back(outputs.back());
93 
94  return outputs;
95  }

103  auto output = graph->zeros({1, dimBatch, dimState});
104  Expr cell = output;

106
