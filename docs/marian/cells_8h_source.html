429  virtual State applyState(std::vector<Expr> xWs,
430  State state,
431  Expr mask = nullptr) {
432  // make sure that we have transition layers
433  assert(transition_ == xWs.empty());
434 
435  auto stateOrig = state.output;
436  auto stateDropped = stateOrig;
437  if(dropMaskS_)
438  stateDropped = dropout(stateOrig, keywords::mask = dropMaskS_);
439 
440  Expr sU;
441  if(layerNorm_) {
442  Expr U; // Temp_1_ in Amun
443  Expr Ux; // Temp_2_ in Amun
444 
445  if(encoder_) {
446  U = layer_norm(dot(stateDropped, U_), U_lns_, U_lnb_, NEMATUS_LN_EPS);
447  Ux = layer_norm(
448  dot(stateDropped, Ux_), Ux_lns_, Ux_lnb_, NEMATUS_LN_EPS);
449 
450  if(transition_) {
451  U = U + b_;
452  }
453  } else {
454  if(final_ || transition_) {
455  U = affine(stateDropped, U_, b_);
456  Ux = affine(stateDropped, Ux_, bx_);
457  } else {
458  U = dot(stateDropped, U_);
459  Ux = dot(stateDropped, Ux_);
460  }
461  U = layer_norm(U, U_lns_, U_lnb_, NEMATUS_LN_EPS);
462  Ux = layer_norm(Ux, Ux_lns_, Ux_lnb_, NEMATUS_LN_EPS);
463  }
464 
465  sU = concatenate({U, Ux}, keywords::axis = -1);
466  } else {
467  sU = dot(stateDropped, UUx_);
468  }
469 
470  Expr xW;
471  if(transition_) {
472  if(!fakeInput_ || fakeInput_->shape() != sU->shape())
473  fakeInput_
474  = sU->graph()->constant(sU->shape(), keywords::init = inits::zeros);
475  xW = fakeInput_;
476  } else {
477  xW = xWs.front();
478  }
479 
480  Expr output;
481  output = mask ? gruOps({stateOrig, xW, sU, bbx_, mask}, final_)
482  : gruOps({stateOrig, xW, sU, bbx_}, final_);

647  ABORT_IF(inputs.empty(), "Multiplicative LSTM expects input");
