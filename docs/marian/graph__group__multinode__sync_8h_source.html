164  void loadDeviceConfig(std::vector<size_t> deviceConfig) {
165  size_t index = 0, node = 0, nClientsSeen = 0;
166  numberClientsOfNodes_ = std::vector<int>(mpi_comm_world_size_, 0);
167  while(index < deviceConfig.size()) {
168  if(numberClientsOfNodes_[node] == 0) {
169  numberClientsOfNodes_[node] = deviceConfig[index];
170  nClientsSeen = 0;
171  } else if(nClientsSeen < numberClientsOfNodes_[node]) {
172  if(node == mpi_my_rank_) {
173  devices_.push_back(deviceConfig[index]);
174  }
175  nClientsSeen++;
176  } else {
177  node++;
178  index--;
179  }
180  index++;
181  }
182  }
