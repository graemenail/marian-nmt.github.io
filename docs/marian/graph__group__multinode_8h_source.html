360  void loadDeviceConfig(std::vector<size_t> deviceConfig) {
361  size_t index = 0, node = 0, nClientsSeen = 0;
362  numberClientsOfNodes_ = std::vector<int>(mpi_comm_world_size_, 0);
363  while(index < deviceConfig.size()) {
364  if(numberClientsOfNodes_[node] == 0) {
365  numberClientsOfNodes_[node] = deviceConfig[index];
366  nClientsSeen = 0;
367  } else if(nClientsSeen < numberClientsOfNodes_[node]) {
368  if(node == mpi_my_rank_) {
369  devices_.push_back(deviceConfig[index]);
370  }
371  nClientsSeen++;
372  } else {
373  node++;
374  index--;
375  }
376  index++;
377  }
378  }
