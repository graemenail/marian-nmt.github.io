306  Expr param(const std::string& pname,
307  const Shape& shape,
308  const NodeInitializer& init,
309  bool fixed = false) {
310  std::string name = pname;
311  if(!namespace_.empty())
312  name = namespace_ + "::" + name;
313 
314  // check first if parameter already exists
315  auto p = params_->get(name);
316  if(p) {
317  // if yes add to tape and return
318  ABORT_IF(shape != p->shape(),
319  "Requested shape {} for existing parameter '{}' does not match "
320  "original shape {}",
321  shape,
322  name,
323  p->shape());
324 
325  p->setTrainable(!fixed);
326  add(p);
327  return p;
328  }
329 
330  // if graph was reloaded do not allow creation of new parameters
331  ABORT_IF(reloaded_,
332  "Graph was reloaded and parameter '{}' is newly created",
333  name);
334 
335  // if not check if name is not taken by other node
336  ABORT_IF(get(name), "Non-parameter with name '{}' already exists", name);
337 
338  // create parameter node (adds to tape)
339  p = Expression<ParamNode>(shared_from_this(), shape, init, fixed);
340 
341  // add to list of parameters
342  p->set_name(name);
343  params_->add(p, name);
344  return p;
