66  virtual Ptr<DecoderState> startState(
67  Ptr<ExpressionGraph> graph,
68  Ptr<data::CorpusBatch> batch,
69  std::vector<Ptr<EncoderState>>& encStates) {
70  using namespace keywords;
71 
72  std::vector<Expr> meanContexts;
73  for(auto& encState : encStates) {
74  // average the source context weighted by the batch mask
75  // this will remove padded zeros from the average
76  meanContexts.push_back(weighted_average(
77  encState->getContext(), encState->getMask(), axis = -3));
78  }
79 
80  Expr start;
81  if(!meanContexts.empty()) {
82  // apply single layer network to mean to map into decoder space
83  auto mlp = mlp::mlp(graph) //
84  .push_back(mlp::dense(graph) //
85  ("prefix", prefix_ + "_ff_state") //
86  ("dim", opt<int>("dim-rnn")) //
87  ("activation", mlp::act::tanh) //
88  ("layer-normalization",
89  opt<bool>("layer-normalization")));
90  start = mlp->apply(meanContexts);
91  }
92 
93  rnn::States startStates(opt<size_t>("dec-depth"), {start, start});
94  return New<DecoderStateHardAtt>(
95  startStates, nullptr, encStates, std::vector<size_t>({0}));
96  }

98  virtual Ptr<DecoderState> step(Ptr<ExpressionGraph> graph,
99  Ptr<DecoderState> state) {
100  using namespace keywords;
101 
102  auto type = options_->get<std::string>("type");
103 
104  int dimTrgVoc = options_->get<std::vector<int>>("dim-vocabs").back();
105 
106  int dimTrgEmb = options_->get<int>("dim-emb");
107 
108  int dimDecState = options_->get<int>("dim-rnn");
109  bool layerNorm = options_->get<bool>("layer-normalization");
110  bool skipDepth = options_->get<bool>("skip");
111 
112  size_t decoderLayers = options_->get<size_t>("dec-depth");
113  auto cellType = options_->get<std::string>("dec-cell");
114 
115  float dropoutRnn = inference_ ? 0 : options_->get<float>("dropout-rnn");
116  float dropoutTrg = inference_ ? 0 : options_->get<float>("dropout-trg");
117 
118  auto stateHardAtt = std::dynamic_pointer_cast<DecoderStateHardAtt>(state);
119 
120  auto trgEmbeddings = stateHardAtt->getTargetEmbeddings();
121 
122  auto context = stateHardAtt->getEncoderStates()[0]->getContext();
123  int dimContext = context->shape()[-1];
124  int dimSrcWords = context->shape()[-3];
125 
126  int dimBatch = context->shape()[-2];
127  int dimTrgWords = trgEmbeddings->shape()[-3];
128  int dimBeam = trgEmbeddings->shape()[-4];
129 
130  if(dropoutTrg) {
131  auto trgWordDrop = graph->dropout(dropoutTrg, {dimTrgWords, dimBatch, 1});
132  trgEmbeddings = dropout(trgEmbeddings, mask = trgWordDrop);
133  }
134 
135  auto flatContext = reshape(context, {dimBatch * dimSrcWords, dimContext});
136  auto attendedContext
137  = rows(flatContext, stateHardAtt->getAttentionIndices());
138  attendedContext = reshape(attendedContext,
139  {dimBeam, dimTrgWords, dimBatch, dimContext});
140 
141  auto rnnInputs = concatenate({trgEmbeddings, attendedContext}, axis = -1);
142  int dimInput = rnnInputs->shape()[-1];
143 
144  if(!rnn_) {
145  auto rnn = rnn::rnn(graph) //
146  ("type", cellType) //
147  ("dimInput", dimInput) //
148  ("dimState", dimDecState) //
149  ("dropout", dropoutRnn) //
150  ("layer-normalization", layerNorm) //
151  ("skip", skipDepth);
152 
153  if(type == "hard-soft-att") {
154  auto attCell = rnn::stacked_cell(graph) //
155  .push_back(rnn::cell(graph) //
156  ("prefix", prefix_ + "_cell1"));
157  for(int i = 0; i < state->getEncoderStates().size(); ++i) {
158  std::string prefix = prefix_;
159  if(state->getEncoderStates().size() > 1)
160  prefix += "_att" + std::to_string(i + 1);
161 
162  attCell.push_back(rnn::attention(graph) //
163  ("prefix", prefix) //
164  .set_state(state->getEncoderStates()[i]));
165  }
166 
167  attCell.push_back(rnn::cell(graph) //
168  ("prefix", prefix_ + "_cell2") //
169  ("final", true));
170  rnn.push_back(attCell);
171  } else {
172  rnn.push_back(rnn::cell(graph)("prefix", prefix_));
173  }
174 
175  for(int i = 0; i < decoderLayers - 1; ++i)
176  rnn.push_back(rnn::cell(graph) //
177  ("prefix", prefix_ + "_l" + std::to_string(i)));
178 
179  rnn_ = rnn.construct();
180  }
181 
182  auto decContext = rnn_->transduce(rnnInputs, stateHardAtt->getStates());
183  rnn::States decStates = rnn_->lastCellStates();
184 
186  auto out = mlp::mlp(graph)
187  .push_back(mlp::dense(graph) //
188  ("prefix", prefix_ + "_ff_logit_l1") //
189  ("dim", dimTrgEmb) //
190  ("activation", mlp::act::tanh) //
191  ("layer-normalization", layerNorm)) //
192  .push_back(mlp::dense(graph) //
193  ("prefix", prefix_ + "_ff_logit_l2") //
194  ("dim", dimTrgVoc));
195 
196  Expr logits;
197  if(type == "hard-soft-att") {
198  std::vector<Expr> alignedContexts;
199  for(int k = 0; k < state->getEncoderStates().size(); ++k) {
200  // retrieve all the aligned contexts computed by the attention mechanism
201  auto att = rnn_->at(0)
202  ->as<rnn::StackedCell>()
203  ->at(k + 1)
204  ->as<rnn::Attention>();
205  alignedContexts.push_back(att->getContext());
206  }
207 
208  Expr alignedContext;
209  if(alignedContexts.size() > 1)
210  alignedContext = concatenate(alignedContexts, axis = -1);
211  else if(alignedContexts.size() == 1)
212  alignedContext = alignedContexts[0];
213 
214  logits = out->apply(rnnInputs, decContext, alignedContext);
215  } else {
216  logits = out->apply(rnnInputs, decContext);
217  }
218 
219  return New<DecoderStateHardAtt>(decStates,
220  logits,
221  stateHardAtt->getEncoderStates(),
222  stateHardAtt->getAttentionIndices());
223  }
