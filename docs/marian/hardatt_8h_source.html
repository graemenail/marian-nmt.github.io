203  virtual std::tuple<Expr, Expr> groundTruth(Ptr<DecoderState> state,
204  Ptr<ExpressionGraph> graph,
205  Ptr<data::CorpusBatch> batch,
206  size_t index) {
207  using namespace keywords;
208 
209  auto ret = DecoderBase::groundTruth(state, graph, batch, index);
210 
211  auto subBatch = (*batch)[index];
212  int dimBatch = subBatch->batchSize();
213  int dimWords = subBatch->batchWidth();
214 
215  std::vector<size_t> attentionIndices(dimBatch, 0);
216  std::vector<size_t> currentPos(dimBatch, 0);
217  std::iota(currentPos.begin(), currentPos.end(), 0);
218 
219  for(int i = 0; i < dimWords - 1; ++i) {
220  for(int j = 0; j < dimBatch; ++j) {
221  size_t word = subBatch->indeces()[i * dimBatch + j];
222  if(specialSymbols_.count(word))
223  currentPos[j] += dimBatch;
224  attentionIndices.push_back(currentPos[j]);
225  }
226  }
227 
228  std::dynamic_pointer_cast<DecoderStateHardAtt>(state)->setAttentionIndices(
229  attentionIndices);
230 
231  return ret;
232  }
