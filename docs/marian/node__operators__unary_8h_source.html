75  : NaryNodeOp(nodes, keywords::shape = newShape(nodes)) {}

77  Shape newShape(const std::vector<Expr>& nodes) {
78  Shape shape = nodes[0]->shape();
79 
80  for(int n = 1; n < nodes.size(); ++n) {
81  Shape shapen = nodes[n]->shape();
82  for(int i = 0; i < shapen.size(); ++i) {
83  UTIL_THROW_IF2(shape[i] != shapen[i] && shape[i] != 1 && shapen[i] != 1,
84  "Shapes cannot be broadcasted");
85  shape.set(i, std::max(shape[i], shapen[i]));
86  }
87  }
88  return shape;
89  }

91  NodeOps forwardOps() {
92  switch(children_.size()) {
93  case 1: return {NodeOp(Element(_1 = Tanh(_2), val_, child(0)->val()))};
94  case 2:
95  return {NodeOp(Element(
96  _1 = Tanh(_2 + _3), val_, child(0)->val(), child(1)->val()))};
97  case 3:
98  return {NodeOp(Element(_1 = Tanh(_2 + _3 + _4),
99  val_,
100  child(0)->val(),
101  child(1)->val(),
102  child(2)->val()))};
103  default:
104  return {
105  NodeOp(Element(_1 = _2 + _3 + _4,
106  val_,
107  child(0)->val(),
108  child(1)->val(),
109  child(2)->val());
110  for(int i = 3; i < children_.size(); ++i)
111  Element(_1 += _2, val_, child(i)->val());
112  Element(_1 = Tanh(_1), val_);)
113  };
114  }
115  }

190  NodeOps backwardOps() {
191  // For each row, the Jacobian times vector is given by:
192  // J * dy = p .* (dy - avg*1)
193  // where avg = p'*dy and p is the softmax output (probabilities).
194  //
195  // For more information, see sec. 2.5 of the following reference:
196  // André F. T. Martins and Ramon Astudillo.
197  // "From Softmax to Sparsemax: A Sparse Model of Attention and Multi-Label
198  // Classification." ICML 2016.
199  // http://jmlr.org/proceedings/papers/v48/martins16.pdf
200 
201  // val_ is already masked if there is a mask, so no need to apply here.
202 
203  return {NodeOp(SoftmaxGrad(child(0)->grad(), adj_, val_))};
204  }
205 
206  const std::string type() { return "softmax"; }
