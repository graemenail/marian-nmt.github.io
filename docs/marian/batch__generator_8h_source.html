48  void fillBatches(bool shuffle = true) {
49  typedef typename sample::value_type Item;
50  auto itemCmp
51  = [](const Item& sa, const Item& sb) { return sa.size() < sb.size(); };
52 
53  auto cmpSrc = [itemCmp](const sample& a, const sample& b) {
54  return std::lexicographical_compare(
55  a.begin(), a.end(), b.begin(), b.end(), itemCmp);
56  };
57 
58  auto cmpTrg = [itemCmp](const sample& a, const sample& b) {
59  return std::lexicographical_compare(
60  a.rbegin(), a.rend(), b.rbegin(), b.rend(), itemCmp);
61  };
62 
63  auto cmpNone = [](const sample& a, const sample& b) { return &a < &b; };
64 
65  typedef std::function<bool(const sample&, const sample&)> cmp_type;
66  typedef std::priority_queue<sample, samples, cmp_type> sample_queue;
67 
68  std::unique_ptr<sample_queue> maxiBatch;
69 
70  if(options_->has("maxi-batch-sort")) {
71  if(options_->get<std::string>("maxi-batch-sort") == "src")
72  maxiBatch.reset(new sample_queue(cmpSrc));
73  else if(options_->get<std::string>("maxi-batch-sort") == "none")
74  maxiBatch.reset(new sample_queue(cmpNone));
75  else
76  maxiBatch.reset(new sample_queue(cmpTrg));
77  } else {
78  maxiBatch.reset(new sample_queue(cmpNone));
79  }
80 
81  int maxBatchSize = options_->get<int>("mini-batch");
82  int maxSize = maxBatchSize * options_->get<int>("maxi-batch");
83 
84  // consume data from corpus into maxi-batch (single sentences)
85  // sorted into specified order (due to queue)
86  size_t sets = 0;
87  while(current_ != data_->end() && maxiBatch->size() < maxSize) {
88  maxiBatch->push(*current_);
89  sets = current_->size();
90  current_++;
91  }
92 
93  samples batchVector;
94  int currentWords = 0;
95  std::vector<size_t> lengths(sets, 0);
96 
97  std::vector<BatchPtr> tempBatches;
98 
99  // while there are sentences in the queue
100  while(!maxiBatch->empty()) {
101  // push item onto batch
102  batchVector.push_back(maxiBatch->top());
103  currentWords += batchVector.back()[0].size();
104  maxiBatch->pop();
105 
106  // Batch size based on sentences
107  bool makeBatch = batchVector.size() == maxBatchSize;
108 
109  // Batch size based on words
110  if(options_->has("mini-batch-words")) {
111  int mbWords = options_->get<int>("mini-batch-words");
112  if(mbWords > 0)
113  makeBatch = currentWords > mbWords;
114  }
115 
116  if(options_->has("mini-batch-fit")) {
117  // Dynamic batching
118  if(stats_) {
119  for(size_t i = 0; i < sets; ++i)
120  if(batchVector.back()[i].size() > lengths[i])
121  lengths[i] = batchVector.back()[i].size();
122 
123  maxBatchSize = stats_->getBatchSize(lengths);
124 
125  if(batchVector.size() > maxBatchSize) {
126  maxiBatch->push(batchVector.back());
127  batchVector.pop_back();
128  makeBatch = true;
129  } else {
130  makeBatch = batchVector.size() == maxBatchSize;
131  }
132  }
133  }
134 
135  // if batch has desired size create a real batch
136  if(makeBatch) {
137  tempBatches.push_back(data_->toBatch(batchVector));
138 
139  // prepare for next batch
140  batchVector.clear();
141  currentWords = 0;
142  lengths.clear();
143  lengths.resize(sets, 0);
144  }
145  }
146 
147  // turn rest into batch
148  if(!batchVector.empty())
149  tempBatches.push_back(data_->toBatch(batchVector));
150 
151  if(shuffle) {
152  // shuffle the batches
153  std::shuffle(tempBatches.begin(), tempBatches.end(), eng_);
154  }
155 
156  // put batches onto queue
157  // exclusive lock
158  std::unique_lock<std::mutex> lock(loadMutex_);
159  for(const auto& batch : tempBatches)
160  bufferedBatches_.push_back(batch);
161  }

181  loadCondition_.wait(
