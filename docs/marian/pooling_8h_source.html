28 Expr MeanInTime(Ptr<ExpressionGraph> graph, Expr x, int k) {
29  using namespace keywords;
30  int dimBatch = x->shape()[0];
31  int dimInput = x->shape()[1];
32  int dimSrcWords = x->shape()[2];

34  auto padding = graph->zeros(shape={dimBatch, dimInput, k / 2});
35  auto xpad = concatenate({padding, x, padding}, axis=2);
36 
37  std::vector<Expr> means;
38  for(int i = 0; i < dimSrcWords; ++i) {

48 Expr ConvolutionInTime(Ptr<ExpressionGraph> graph, Expr x,
49  int k, std::string name) {
50  using namespace keywords;
51  int dimBatch = x->shape()[0];
52  int dimInput = x->shape()[1];
53  int dimSrcWords = x->shape()[2];
54 
55  auto padding = graph->zeros(shape={dimBatch, dimInput, k / 2});
56  auto xpad = concatenate({padding, x, padding}, axis=2);
57 
58  float scale = 1.f / sqrtf(k * dimInput);
59  auto K = graph->param(name, {1, dimInput, k}, keywords::init=inits::uniform(scale));
60  auto B = graph->param(name + "_b", {1, dimInput}, init=inits::zeros);
61 
62  std::vector<Expr> filters;
63  for(int i = 0; i < dimSrcWords; ++i) {
64  std::vector<Expr> preAvg;
65  for(int j = 0; j < k; ++j)
66  preAvg.push_back(step(xpad, i + j));
67 
68  filters.push_back(sum(concatenate(preAvg, axis=2) * K, axis=2));
69  }
70  return tanh(concatenate(filters, axis=2), B, x);
71 }
