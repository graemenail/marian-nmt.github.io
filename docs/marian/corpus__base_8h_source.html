169  size_t subSize = std::ceil(size_ / (float)n);
170  size_t totSize = size_;
171 
172  int pos = 0;
173  for(int k = 0; k < n; ++k) {
174  size_t __size__ = std::min(subSize, totSize);
175 
176  auto sb = New<SubBatch>(__size__, width_);
177 
178  size_t __words__ = 0;
179  for(int j = 0; j < width_; ++j) {
180  for(int i = 0; i < __size__; ++i) {
181  sb->data()[j * __size__ + i] = indices_[j * size_ + pos + i];
182  sb->mask()[j * __size__ + i] = mask_[j * size_ + pos + i];
183 
184  if(mask_[j * size_ + pos + i] != 0)
185  __words__++;
186  }
187  }
188 
189  sb->setWords(__words__);
190  splits.push_back(sb);
191 
192  totSize -= __size__;
193  pos += __size__;
194  }
