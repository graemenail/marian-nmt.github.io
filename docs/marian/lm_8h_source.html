117  // in order to continue decoding for the next word
118  rnn::States decoderStates = rnn_->lastCellStates();

121  auto layer1 = mlp::dense(graph)
122  ("prefix", prefix_ + "_ff_logit_l1")
123  ("dim", opt<int>("dim-emb"))
124  ("activation", mlp::act::tanh)
