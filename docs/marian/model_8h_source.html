68  virtual Expr construct(Ptr<ExpressionGraph> g,
69  Ptr<data::Batch> batch,
70  bool inference = false) {
71  using namespace keywords;
72  const std::vector<int> dims = {784, 2048, 2048, 10};
73 
74  // Start with an empty expression graph
75  clear(g);
76 
77  // Create an input layer of shape batchSize x numFeatures and populate it
78  // with training features
79  auto features
80  = std::static_pointer_cast<data::DataBatch>(batch)->features();
81  auto x = g->constant({(int)batch->size(), dims[0]},
82  inits::from_vector(features));
83 
84  // Construct hidden layers
85  std::vector<Expr> layers, weights, biases;
86 
87  for(size_t i = 0; i < dims.size() - 1; ++i) {
88  int in = dims[i];
89  int out = dims[i + 1];
90 
91  if(i == 0) {
92  // Create a dropout node as the parent of x,
93  // and place that dropout node as the value of layers[0]
94  layers.emplace_back(dropout(x, 0.2));
95  } else {
96  // Multiply the matrix in layers[i-1] by the matrix in weights[i-1]
97  // Take the result, and perform matrix addition on biases[i-1].
98  // Wrap the result in rectified linear activation function,
99  // and finally wrap that in a dropout node
100  layers.emplace_back(dropout(
101  relu(affine(layers.back(), weights.back(), biases.back())), 0.5));
102  }
103 
104  // Construct a weight node for the outgoing connections from layer i
105  weights.emplace_back(
106  g->param("W" + std::to_string(i), {in, out}, inits::uniform()));
107 
108  // Construct a bias node. These weights are initialized to zero
109  biases.emplace_back(
110  g->param("b" + std::to_string(i), {1, out}, inits::zeros));
111  }
112 
113  // Perform matrix multiplication and addition for the last layer
114  auto last = affine(layers.back(), weights.back(), biases.back());
115 
116  if(!inference) {
117  // Create an output layer of shape batchSize x 1 and populate it with
118  // labels
119  auto labels = std::static_pointer_cast<data::DataBatch>(batch)->labels();
120  auto y = g->constant({(int)batch->size(), 1}, inits::from_vector(labels));
121 
122  // Define a top-level node for training
123  return mean(cross_entropy(last, y), axis = 0);
124  } else {
125  // Define a top-level node for inference
126  return logsoftmax(last);
127  }
128  }
129 };
